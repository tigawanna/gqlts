## Typed GraphQL as code

```ts
import { createClient, everything } from './generated';

const client = createClient({
  url: 'https://countries.trevorblades.com',
});

const { data } = await client.query({
  countries: [
    { filter: { continent: { eq: 'EU' } } },
    {
      code: true,
      name: true,
      languages: {
        ...everything,
      },
    },
  ],
  __alias: {
    featuredCountry: {
      country: [{ code: 'CA' }, { name: true, emoji: true }],
    },
  },
});

data?.countries[0].languages[0].name;
data?.featuredCountry?.emoji;
```

The generated operation stays ordinary GraphQL:

```graphql
query ($v1: CountryFilterInput, $v2: ID!) {
  countries(filter: $v1) {
    code
    name
    languages {
      code
      name
      native
      rtl
    }
  }
  featuredCountry: country(code: $v2) {
    name
    emoji
  }
}
```
